    <script type="module">
        import { initializeApp } from "https://www.gstatic.com/firebasejs/10.7.1/firebase-app.js";
        import { getAuth, GoogleAuthProvider, signInWithPopup, onAuthStateChanged, signOut } from "https://www.gstatic.com/firebasejs/10.7.1/firebase-auth.js";
        import { getFirestore, collection, addDoc, getDocs, query, where, deleteDoc, doc } from "https://www.gstatic.com/firebasejs/10.7.1/firebase-firestore.js";

        // CONFIGURACIÓN REVISADA
        const firebaseConfig = {
            apiKey: "AIzaSyDsERd3f0oOGOyrShHIrixFaipSojKJc8g",
            authDomain: "marketplace-da7a4.firebaseapp.com",
            projectId: "marketplace-da7a4",
            storageBucket: "marketplace-da7a4.firebasestorage.app",
            messagingSenderId: "978612277776",
            appId: "1:978612277776:web:6dabc86e700efc4d067095"
        };

        const app = initializeApp(firebaseConfig);
        const auth = getAuth(app);
        const db = getFirestore(app);
        const provider = new GoogleAuthProvider();

        let currentUser = null;
        let allProducts = [];

        window.showView = (name) => {
            document.querySelectorAll('section').forEach(s => s.classList.add('hidden'));
            document.getElementById(`view-${name}`).classList.remove('hidden');
            if(name === 'comprador') loadCatalog();
            if(name === 'emprendedor') loadMyProducts();
        };

        window.setRole = (role) => {
            if(role === 'emprendedor' && !currentUser) {
                loginGoogle();
            } else {
                showView(role);
            }
        };

        window.loginGoogle = async () => {
            try {
                const result = await signInWithPopup(auth, provider);
                currentUser = result.user;
                showView('emprendedor');
            } catch (e) {
                console.error("Error en login:", e);
                alert("Error al iniciar sesión. Revisa que el dominio esté autorizado en la consola de Firebase.");
            }
        };

        window.logout = () => signOut(auth).then(() => location.reload());

        onAuthStateChanged(auth, (user) => {
            if (user) {
                currentUser = user;
                document.getElementById('auth-buttons').classList.add('hidden');
                document.getElementById('user-info').classList.remove('hidden');
                document.getElementById('user-name').innerText = user.displayName;
            }
        });

        const form = document.getElementById('form-product');
        form.addEventListener('submit', async (e) => {
            e.preventDefault();
            try {
                await addDoc(collection(db, "products"), {
                    name: document.getElementById('p-name').value,
                    desc: document.getElementById('p-desc').value,
                    grade: document.getElementById('p-grade').value,
                    cat: document.getElementById('p-cat').value,
                    price: document.getElementById('p-price').value,
                    contact: document.getElementById('p-contact').value,
                    img: document.getElementById('p-img').value,
                    userId: currentUser.uid
                });
                form.reset();
                closeModal('modal-add');
                loadMyProducts();
            } catch (error) {
                console.error("Error al guardar:", error);
            }
        });

        window.loadCatalog = async () => {
            const snap = await getDocs(collection(db, "products"));
            allProducts = [];
            snap.forEach(d => allProducts.push({id: d.id, ...d.data()}));
            render(allProducts);
        };

        const render = (data) => {
            const container = document.getElementById('catalog-container');
            container.innerHTML = "";
            data.forEach(p => {
                container.innerHTML += `
                    <div class="bg-white rounded-xl shadow-lg p-4 hover:shadow-2xl transition">
                        <img src="${p.img}" class="w-full h-40 object-cover rounded-lg mb-3">
                        <h4 class="font-bold text-navyBlue text-lg">${p.name}</h4>
                        <p class="text-xs text-turquesa font-bold uppercase mb-2">${p.cat} - ${p.grade}</p>
                        <p class="text-sm text-gray-600 mb-4 h-10 overflow-hidden">${p.desc}</p>
                        <div class="flex justify-between items-center border-t pt-3">
                            <span class="font-bold text-xl text-navyBlue">$${p.price}</span>
                            <a href="https://wa.me/${p.contact}" target="_blank" class="bg-navyBlue text-white px-4 py-2 rounded-lg text-sm font-bold hover:bg-turquesa transition">Pedir</a>
                        </div>
                    </div>`;
            });
        };

        window.loadMyProducts = async () => {
            if(!currentUser) return;
            const q = query(collection(db, "products"), where("userId", "==", currentUser.uid));
            const snap = await getDocs(q);
            const container = document.getElementById('my-products-container');
            container.innerHTML = "";
            snap.forEach(d => {
                const p = d.data();
                container.innerHTML += `
                    <div class="flex justify-between items-center bg-white p-5 rounded-xl shadow border-l-8 border-turquesa">
                        <div>
                            <p class="font-bold text-navyBlue">${p.name}</p>
                            <p class="text-sm text-gray-500">$${p.price}</p>
                        </div>
                        <button onclick="deleteP('${d.id}')" class="text-red-500 hover:text-red-700 font-bold">Eliminar</button>
                    </div>`;
            });
        };

        window.deleteP = async (id) => {
            if(confirm("¿Estás seguro de eliminar este producto?")) { 
                await deleteDoc(doc(db, "products", id)); 
                loadMyProducts(); 
            }
        };

        window.openModal = (id) => document.getElementById(id).classList.remove('hidden');
        window.closeModal = (id) => document.getElementById(id).classList.add('hidden');
        window.filterProducts = () => {
            const t = document.getElementById('search-input').value.toLowerCase();
            const c = document.getElementById('category-filter').value;
            render(allProducts.filter(p => (p.name.toLowerCase().includes(t)) && (c === 'todos' || p.cat === c)));
        };
    </script>
